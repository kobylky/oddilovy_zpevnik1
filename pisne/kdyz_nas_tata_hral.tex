\beginsong{Když náš táta hrál}[by={J. Cash, J. Vyčítal}]
\beginverse
\chordson
Když \[D]jsem byl chlapec malej, tak metr nad zemí,
\[G]scházeli se farmáři tam \[D]u nás v přízemí,
\chordsoff
mezi nima můj táta u piva sedával
\chordson
\[A7]a tu svoji nejmilejší \[D]hrál.
\endverse

kdybych si ale ve světě moh' ještě něco přát,
tak zase slyšet svýho tátu hrát.
\endverse

když jsem si vzpomněl, jak náš táta hrál.
\endverse

zase jeho píseň slyšel hrát.
\endverse
