\beginsong{Hejno vran}[by={Wabi Ryvola}]

\beginverse
\chordson
\[G]Když nad pasekou vzlétne \[C]hejno vran,
\[Cmi]hejno vran, \[G]hejno vran,
je léta pel už dávno \[C]vyprodán,
\[D7]podzim se hlásí chladem \[G]jasných rán,
v tom ránu prochází se \[C]sám a sám,
\[D7]kdo městu dávno sbohem \[G]dal.\[C]
\endverse

\beginrefrain
\chordson
\[G]A tak se toulá trávou \[H7]spálenou,
neví jak rozloučit se s \[Emi]létem,
dívá se na krajinu \[C]vzdálenou,
\[D7]které dal celý život \[G]svůj.\[C]
\endrefrain

\beginverse
\chordsoff
Nad hlavou zakrouží ti hejno vran,
hejno vran, hejno vran,
a bílých hvězd je všude oceán,
přece si nepostavíš pod ně stan,
budeš spát pod širákem, jak tě znám,
i když už podzim přichází.
\endverse

\beginrefrain
\chordsoff

\endrefrain
